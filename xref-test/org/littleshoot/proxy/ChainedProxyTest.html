15  /**
16   * Tests a proxy chained to a downstream proxy. In addition to the usual
17   * assertions, this also asserts that every request sent by the upstream proxy
18   * was received by the downstream proxy.
19   */
20  public class ChainedProxyTest extends BaseProxyTest {
21      protected static final AtomicInteger DOWNSTREAM_PROXY_SERVER_PORT_SEQ = new AtomicInteger(
22              59000);
23  
24      private int downstreamProxyPort;
25      private String downstreamProxyHostAndPort;
26  
27      private static final AtomicLong REQUESTS_SENT_BY_UPSTREAM = new AtomicLong(
28              0l);
29      private static final AtomicLong REQUESTS_RECEIVED_BY_DOWNSTREAM = new AtomicLong(
30              0l);
31      private static final ConcurrentSkipListSet<TransportProtocol> TRANSPORTS_USED = new ConcurrentSkipListSet<TransportProtocol>();
32  
33      private static final ActivityTracker UPSTREAM_TRACKER = new ActivityTrackerAdapter() {
34          public void requestSent(FlowContext flowContext,
35                  io.netty.handler.codec.http.HttpRequest httpRequest) {
36              REQUESTS_SENT_BY_UPSTREAM.incrementAndGet();
37              TRANSPORTS_USED.add(flowContext.getOutboundTransportProtocol());
38          }
39      };
40      private static final ActivityTracker DOWNSTREAM_TRACKER = new ActivityTrackerAdapter() {
41          public void requestReceivedFromClient(FlowContext flowContext,
42                  io.netty.handler.codec.http.HttpRequest httpRequest) {
43              REQUESTS_RECEIVED_BY_DOWNSTREAM.incrementAndGet();
44          };
45      };
46  
47      private HttpProxyServer downstreamProxy;
48  
49      @Override
50      protected void setUp() {
51          // Set up ports from sequence
52          downstreamProxyPort = DOWNSTREAM_PROXY_SERVER_PORT_SEQ
53                  .getAndIncrement();
54          downstreamProxyHostAndPort = "127.0.0.1:" + downstreamProxyPort;
55  
56          REQUESTS_SENT_BY_UPSTREAM.set(0);
57          REQUESTS_RECEIVED_BY_DOWNSTREAM.set(0);
58          TRANSPORTS_USED.clear();
59          final SSLContextSource sslContextSource = new SelfSignedSSLContextSource(
60                  "chain_proxy_keystore_1.jks");
61          this.downstreamProxy = DefaultHttpProxyServer.bootstrap()
62                  .withName("DownstreamProxy")
63                  .withTransportProtocol(UDT)
64                  .withSslContextSource(sslContextSource)
65                  .create().listen(downstreamProxyPort);
66          this.downstreamProxy.addActivityTracker(DOWNSTREAM_TRACKER);
67          this.proxyServer = DefaultHttpProxyServer.bootstrap()
68                  .withChainProxyManager(new ChainedProxyManagerAdapter() {
69                      public String getHostAndPort(HttpRequest httpRequest) {
70                          return downstreamProxyHostAndPort;
71                      }
72  
73                      @Override
74                      public TransportProtocol getTransportProtocol() {
75                          return TransportProtocol.UDT;
76                      }
77  
78                      @Override
79                      public boolean requiresEncryption(HttpRequest httpRequest) {
80                          return true;
81                      }
82  
83                      @Override
84                      public SSLContext getSSLContext() {
85                          return sslContextSource.getSSLContext();
86                      }
87                  }).create().listen(proxyServerPort);
88          this.proxyServer.addActivityTracker(UPSTREAM_TRACKER);
89      }
90  
91      @Override
92      protected void tearDown() throws Exception {
93          this.downstreamProxy.stop();
94      }

114     private void assertThatDownstreamProxyReceivedSentRequests() {
115         Assert.assertEquals(
116                 "Downstream proxy should have seen every request sent by upstream proxy",
117                 REQUESTS_SENT_BY_UPSTREAM.get(),
118                 REQUESTS_RECEIVED_BY_DOWNSTREAM.get());
119         Assert.assertEquals(
120                 "1 and only 1 transport protocol should have been used to downstream proxy",
121                 1, TRANSPORTS_USED.size());
122         Assert.assertTrue("UDT transport should have been used",
123                 TRANSPORTS_USED.contains(TransportProtocol.UDT));
124     }
