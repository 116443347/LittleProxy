115      * given file.

117      * @param path
118      * @return
119      */
120     public static HttpProxyServerBootstrap bootstrapFromFile(String path) {
121         final File propsFile = new File(path);
122         Properties props = new Properties();
123 
124         if (propsFile.isFile()) {
125             InputStream is = null;
126             try {
127                 is = new FileInputStream(propsFile);
128                 props.load(is);
129             } catch (final IOException e) {
130                 LOG.warn("Could not load props file?", e);
131             } finally {
132                 IOUtils.closeQuietly(is);
133             }
134         }
135 
136         return new DefaultHttpProxyServerBootstrap(props);
137     }

264     private void doStart() {
265         ServerBootstrap serverBootstrap = new ServerBootstrap().group(
266                 serverGroup.clientToProxyBossPools.get(transportProtocol),
267                 serverGroup.clientToProxyWorkerPools.get(transportProtocol));
268 
269         ChannelInitializer<Channel> initializer = new ChannelInitializer<Channel>() {
270             protected void initChannel(Channel ch) throws Exception {
271                 new ClientToProxyConnection(
272                         DefaultHttpProxyServer.this,
273                         sslEngineSource,
274                         ch.pipeline());
275             };
276         };
277         switch (transportProtocol) {
278         case TCP:
279             LOG.info("Proxy listening with TCP transport");
280             serverBootstrap.channelFactory(new ChannelFactory<ServerChannel>() {
281                 @Override
282                 public ServerChannel newChannel() {
283                     return new NioServerSocketChannel();
284                 }
285             });
286             break;
287         case UDT:
288             LOG.info("Proxy listening with UDT transport");
289             serverBootstrap.channelFactory(NioUdtProvider.BYTE_ACCEPTOR)
290                     .option(ChannelOption.SO_BACKLOG, 10)
291                     .option(ChannelOption.SO_REUSEADDR, true);
292             break;
293         default:
294             throw new UnknownTransportProtocolError(transportProtocol);
295         }
296         serverBootstrap.childHandler(initializer);
297         serverBootstrap.bind(address).addListener(new ChannelFutureListener() {
298             @Override
299             public void operationComplete(ChannelFuture future)
300                     throws Exception {
301                 registerChannel(future.channel());
302             }
303         }).awaitUninterruptibly();
304     }

340     /**
341      * Represents a group of servers that share thread pools.
342      */
343     private static class ServerGroup {
344         private static final int MAXIMUM_INCOMING_THREADS = 10;
345         private static final int MAXIMUM_OUTGOING_THREADS = 40;
346 
347         /**
348          * A name for this ServerGroup to use in naming threads.
349          */
350         private final String name;
351 
352         /**
353          * Keep track of all channels for later shutdown.
354          */
355         private final ChannelGroup allChannels = new DefaultChannelGroup(
356                 "HTTP-Proxy-Server", GlobalEventExecutor.INSTANCE);
357 
358         /**
359          * These {@link EventLoopGroup}s accept incoming connections to the
360          * proxies. A different EventLoopGroup is used for each
361          * TransportProtocol, since these have to be configured differently.
362          */
363         private final Map<TransportProtocol, EventLoopGroup> clientToProxyBossPools = new ConcurrentHashMap<TransportProtocol, EventLoopGroup>();
364 
365         /**
366          * These {@link EventLoopGroup}s process incoming requests to the
367          * proxies. A different EventLoopGroup is used for each
368          * TransportProtocol, since these have to be configured differently.
369          */
370         private final Map<TransportProtocol, EventLoopGroup> clientToProxyWorkerPools = new ConcurrentHashMap<TransportProtocol, EventLoopGroup>();
371 
372         /**
373          * These {@link EventLoopGroup}s are used for making outgoing
374          * connections to servers. A different EventLoopGroup is used for each
375          * TransportProtocol, since these have to be configured differently.
376          */
377         private final Map<TransportProtocol, EventLoopGroup> proxyToServerWorkerPools = new ConcurrentHashMap<TransportProtocol, EventLoopGroup>();
378 
379         private volatile boolean stopped = false;
380 
381         private ServerGroup(String name) {
382             this.name = name;
383             // Set up worker pools for each transport protocol
384             for (TransportProtocol transportProtocol : TransportProtocol
385                     .values()) {
386                 SelectorProvider selectorProvider = null;
387                 switch (transportProtocol) {
388                 case TCP:
389                     selectorProvider = SelectorProvider.provider();
390                     break;
391                 case UDT:
392                     selectorProvider = NioUdtProvider.BYTE_PROVIDER;
393                     break;
394                 default:
395                     throw new UnknownTransportProtocolError(transportProtocol);
396                 }
397 
398                 this.clientToProxyBossPools.put(transportProtocol,
399                         new NioEventLoopGroup(
400                                 MAXIMUM_INCOMING_THREADS,
401                                 CLIENT_TO_PROXY_THREAD_FACTORY,
402                                 selectorProvider));
403                 this.clientToProxyWorkerPools.put(transportProtocol,
404                         new NioEventLoopGroup(
405                                 MAXIMUM_INCOMING_THREADS,
406                                 CLIENT_TO_PROXY_THREAD_FACTORY,
407                                 selectorProvider));
408                 this.proxyToServerWorkerPools.put(
409                         transportProtocol,
410                         new NioEventLoopGroup(
411                                 MAXIMUM_OUTGOING_THREADS,
412                                 PROXY_TO_SERVER_THREAD_FACTORY,
413                                 selectorProvider));
414             }
415 
416             Thread.setDefaultUncaughtExceptionHandler(new UncaughtExceptionHandler() {
417                 public void uncaughtException(final Thread t, final Throwable e) {
418                     LOG.error("Uncaught throwable", e);
419                 }
420             });
421 
422             Runtime.getRuntime().addShutdownHook(new Thread(new Runnable() {
423                 public void run() {
424                     stop();
425                 }
426             }));
427         }
428 
429         synchronized private void stop() {
430             LOG.info("Shutting down proxy");
431             if (stopped) {
432                 LOG.info("Already stopped");
433                 return;
434             }
435 
436             LOG.info("Closing all channels...");
437 
438             final ChannelGroupFuture future = allChannels.close();
439             future.awaitUninterruptibly(10 * 1000);
440 
441             if (!future.isSuccess()) {
442                 final Iterator<ChannelFuture> iter = future.iterator();
443                 while (iter.hasNext()) {
444                     final ChannelFuture cf = iter.next();
445                     if (!cf.isSuccess()) {
446                         LOG.warn(
447                                 "Unable to close channel.  Cause of failure for {} is {}",
448                                 cf.channel(),
449                                 cf.cause());
450                     }
451                 }
452             }
453 
454             LOG.info("Shutting down event loops");
455             List<EventLoopGroup> allEventLoopGroups = new ArrayList<EventLoopGroup>();
456             allEventLoopGroups.addAll(clientToProxyBossPools.values());
457             allEventLoopGroups.addAll(clientToProxyWorkerPools.values());
458             allEventLoopGroups.addAll(proxyToServerWorkerPools.values());
459             for (EventLoopGroup group : allEventLoopGroups) {
460                 group.shutdownGracefully();
461             }
462 
463             for (EventLoopGroup group : allEventLoopGroups) {
464                 try {
465                     group.awaitTermination(60, TimeUnit.SECONDS);
466                 } catch (InterruptedException ie) {
467                     LOG.warn("Interrupted while shutting down event loop");
468                 }
469             }
470 
471             stopped = true;
472 
473             LOG.info("Done shutting down proxy");
474         }

672         private InetSocketAddress determineListenAddress() {
673             if (address != null) {
674                 return address;
675             } else {
676                 // Binding only to localhost can significantly improve the
677                 // security of
678                 // the proxy.
679                 if (allowLocalOnly) {
680                     return new InetSocketAddress("127.0.0.1", port);
681                 } else if (listenOnAllAddresses) {
682                     return new InetSocketAddress(port);
683                 } else {
684                     try {
685                         return new InetSocketAddress(
686                                 NetworkUtils.getLocalHost(), port);
687                     } catch (final UnknownHostException e) {
688                         LOG.error("Could not get local host?", e);
689                         return new InetSocketAddress(port);
690                     }
691                 }
692             }
